            <!-- Чимэглэл -->

    <script>
        // Од үүсгэх
        const starsContainer = document.getElementById('stars-container');
        for (let i = 0; i < 150; i++) {
            const star = document.createElement('div');
            star.className = 'star';
            const size = Math.random() * 3;
            star.style.width = `${size}px`;
            star.style.height = `${size}px`;
            star.style.top = `${Math.random() * 100}%`;
            star.style.left = `${Math.random() * 100}%`;
            star.style.setProperty('--duration', `${Math.random() * 3 + 2}s`);
            starsContainer.appendChild(star);
        }

        const mainFlower = document.getElementById('mainFlower');
        const initialScene = document.getElementById('initialScene');
        const letterScene = document.getElementById('letterScene');
        const letterBox = document.getElementById('letterBox');
        const yesBtn = document.getElementById('yesBtn');
        const noBtn = document.getElementById('noBtn');

        // Цэцэг дээр дарах
        mainFlower.addEventListener('click', () => {
            initialScene.style.opacity = '0';
            initialScene.style.transform = 'scale(0.8)';
            setTimeout(() => {
                initialScene.classList.add('hidden');
                letterScene.classList.remove('hidden');
                setTimeout(() => {
                    letterScene.classList.add('opacity-100');
                    letterBox.classList.remove('scale-90');
                    letterBox.classList.add('scale-100');
                }, 100);
            }, 500);
        });

        // "Үгүй" товчлуур зугтах
        noBtn.addEventListener('mouseover', () => {
            const x = Math.random() * 200 - 100;
            const y = Math.random() * 200 - 100;
            noBtn.style.transform = `translate(${x}px, ${y}px)`;
        });

        // "Тийм" товчлуур дарах
        yesBtn.addEventListener('click', () => {
            letterBox.style.transform = 'scale(0) rotate(20deg)';
            letterScene.style.opacity = '0';
            
            setTimeout(() => {
                letterScene.classList.add('hidden');
                startFinale();
            }, 700);
        });

        function startFinale() {
            // Салют эффект (Од харвах)
            let count = 0;
            const interval = setInterval(() => {
                createShootingStar();
                count++;
                if (count > 20) clearInterval(interval);
            }, 300);

            // Цэцэгс ургах
            const emojis = ['🌹', '🌸', '🌺', '🌷', '✨', '💖', '🌼'];
            for (let i = 0; i < 50; i++) {
                setTimeout(() => {
                    const flower = document.createElement('div');
                    flower.className = 'flower-bloom text-4xl';
                    flower.style.left = `${Math.random() * 100}%`;
                    flower.innerText = emojis[Math.floor(Math.random() * emojis.length)];
                    document.getElementById('garden').appendChild(flower);
                }, i * 100);
            }

            // Төгсгөлийн бичиг
            setTimeout(() => {
                const msg = document.createElement('div');
                msg.className = 'fixed inset-0 flex items-center justify-center text-white text-4xl font-bold handwritten z-50 animate-pulse text-center px-6';
                msg.innerHTML = "Баярлалаа... ❤️<br><span class='text-xl font-light mt-4 block'>Чи хамгийн шилдэг нь!</span>";
                document.body.appendChild(msg);
             
            }, 2000);
        }

        function createShootingStar() {
            const star = document.createElement('div');
            star.style.position = 'absolute';
            star.style.width = '2px';
            star.style.height = '2px';
            star.style.background = 'white';
            star.style.boxShadow = '0 0 10px 2px white';
            star.style.top = Math.random() * 50 + '%';
            star.style.left = Math.random() * 100 + '%';
            
            const anim = star.animate([
                { transform: 'translate(0, 0) scale(1)', opacity: 1 },
                { transform: 'translate(-500px, 500px) scale(0)', opacity: 0 }
            ], { duration: 1000, easing: 'linear' });

            document.body.appendChild(star);
            anim.onfinish = () => star.remove();
        }
    </script>
